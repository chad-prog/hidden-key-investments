```typescript
import { LeadCreateSchema } from '@/lib/schemas/crm';

const lead = LeadCreateSchema.parse({
  source: 'website',
  contact: {

    phone: '+1234567890'
  },
  property: {
    address: '123 Main St',
    city: 'Austin',
    state: 'TX'

  method: 'POST',
  headers: { 'Content-Type': 'application/json' },

});
```

```bash
    "address": "123 Main St",
    "city": "Austin",
    "state": "TX",
    "zip": "78701",
```
